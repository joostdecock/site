 31:     /**
 32:      * Samples a pattern
 33:      *
 34:      * This samples a pattern, sets the response and sends it
 35:      * Essentially, it takes care of the entire remainder of the request
 36:      *
 37:      * @param \Freesewing\Context $context
 38:      */
 39:     public function run(Context $context)
 40:     {
 41:         $context->addPattern();
 42: 
 43:         if ($context->getChannel()
 44:                 ->isValidRequest($context) === true
 45:         ) :
 46: 
 47:             $context->addUnits();
 48:             $context->getPattern()
 49:                 ->setUnits($context->getUnits());
 50: 
 51:             $context->addTranslator();
 52:             $context->getPattern()->setTranslator($context->getTranslator());
 53: 
 54:             $context->getPattern()->setPartMargin($context->getTheme()->config['settings']['partMargin']);
 55:             $context->getTheme()->setOptions($context->getRequest());
 56: 
 57:             if ($context->getRequest()
 58:                     ->getData('mode') == 'options'
 59:             ) { // Sampling options
 60:                 $context->addOptionsSampler();
 61:                 $context->getOptionsSampler()->setPattern($context->getPattern());
 62: 
 63:                 $context->addModel();
 64:                 $context->getModel()->addMeasurements($context->getOptionsSampler()->loadModelMeasurements($context->getPattern()));
 65: 
 66:                 $context->getPattern()->addOptions($context->getOptionsSampler()->loadPatternOptions());
 67: 
 68:                 $context->setPattern(
 69:                     $context->getOptionsSampler()->sampleOptions(
 70:                         $context->getModel(),
 71:                         $context->getTheme(),
 72:                         $context->getRequest()->getData('option'),
 73:                         $context->getRequest()->getData('steps')
 74:                     )
 75:                 );
 76:             } else { // Sampling measurements
 77:                 $context->addMeasurementsSampler();
 78:                 $context->getMeasurementsSampler()->setPattern($context->getPattern());
 79: 
 80:                 // add options like 'chestEase' to all compare-models
 81:                 $context->getPattern()->addOptions(
 82:                     $context->getChannel()->standardizePatternOptions(
 83:                         $context->getRequest(),
 84:                         $context->getPattern()
 85:                     )
 86:                 );
 87: 
 88:                 $context->getMeasurementsSampler()->setModelConfig($context->getPattern()
 89:                     ->getSamplerModelConfig());
 90:                 $context->getMeasurementsSampler()->loadPatternModels($context->getRequest()
 91:                     ->getData('samplerGroup'));
 92:                 $context->setPattern($context->getMeasurementsSampler()->sampleMeasurements($context->getTheme()));
 93:             }
 94: 
 95:             $context->addSvgDocument();
 96:             $context->addRenderbot();
 97:             $this->svgRender($context);
 98:             $context->setResponse($context->getTheme()
 99:                 ->themeResponse($context));
100:         else: // channel->isValidRequest() !== true
101:             $context->getChannel()
102:                 ->handleInvalidRequest($context);
103:         endif;
