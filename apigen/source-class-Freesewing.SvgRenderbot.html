 26:     /**
 27:      * Returns SVG code for a pattern
 28:      *
 29:      * This does not return the entire SVG document, but merely the SVG body
 30:      * Also note that this mostly just calls renderPart() on all parts
 31:      * that need to be rendered.
 32:      *
 33:      * @param \Freesewing\Pattern or equivalent. The pattern to render
 34:      *
 35:      * @return string The SVG code for the pattern
 36:      */
 37:     public function render($pattern)
 38:     {
 39:         // 90dpi/25.4 = 3.54330709
 40:         $scale = new \Freesewing\Transform('scale', 3.54330709);
 41: 
 42:         $svg = $this->openGroup(
 43:             'patternScaleContainer',
 44:             \Freesewing\Transform::asSvgParameter(array($scale))
 45:         );
 46: 
 47:         if (isset($pattern->parts) && count($pattern->parts) > 0) :
 48:             foreach ($pattern->parts as $partKey => $part) {
 49:                 if ($part->getRender() === true) {
 50:                     $transforms = '';
 51:                     if (is_array($part->transforms) && count($part->transforms) > 0) {
 52:                         $transforms = \Freesewing\Transform::asSvgParameter($part->transforms);
 53:                     }
 54: 
 55:                     $svg .= $this->openGroup($partKey, $transforms);
 56:                     $svg .= $this->renderPart($part);
 57:                     $svg .= $this->closeGroup();
 58:                 }
 59:             }
 60:         endif;
 61: 
 62:         $svg .= $this->closeGroup();
 63: 
 64:         return $svg;
 65:     }

 67:     /**
 68:      * Returns SVG code for a path
 69:      *
 70:      * @param \Freesewing\Path $path The path to render
 71:      * @param \Freesewing\Part $part The part this path is part of (did you get that?)
 72:      *
 73:      * @return string The SVG code for the rendered path
 74:      */
 75:     public function renderPath($path, $part)
 76:     {
 77:         if ($path->getRender() === false) {
 78:             return '';
 79:         }
 80:         $pathstring = $path->getPathstring();
 81:         $points = $part->points;
 82:         $patharray = explode(' ', $pathstring);
 83:         $svg = '';
 84:         foreach ($patharray as $p) {
 85:             $p = rtrim($p);
 86:             if ($p != '' && Utils::isAllowedPathCommand($p)) {
 87:                 $svg .= " $p ";
 88:             } elseif (is_object($points[$p])) {
 89:                 $svg .= ' '.$points[$p]->x.','.$points[$p]->y.' ';
 90:             }
 91:         }
 92:         $attributes = $path->getAttributes();
 93:         if (!isset($attributes['id'])) {
 94:             $attributes['id'] = $this->getUid();
 95:         }
 96: 
 97:         return $this->nl().'<path '.Utils::flattenAttributes($attributes).' d="'.$svg.'" />';
 98:     }

193:     /**
194:      * Returns SVG code for a pattern part
195:      *
196:      * This renders the following elements contained within the part:
197:      *  - includes
198:      *  - paths
199:      *  - snippets
200:      *  - texts
201:      *  - textsonpath
202:      *  - notes
203:      *  - dimensions
204:      *
205:      * @param \Freesewing\Part The part to render
206:      *
207:      * @return The SVG code for the part
208:      */
209:     private function renderPart($part)
210:     {
211:         $svg = $this->nl();
212: 
213:         if (@$part->includes) {
214:             foreach ($part->includes as $include) {
215:                 $svg .= $include->get();
216:             }
217:         }
218:         if (@$part->paths) {
219:             foreach ($part->paths as $path) {
220:                 $svg .= $this->renderPath($path, $part);
221:             }
222:         }
223:         if (@$part->snippets) {
224:             foreach ($part->snippets as $snippet) {
225:                 $svg .= $this->renderSnippet($snippet, $part);
226:             }
227:         }
228:         if (@$part->texts) {
229:             foreach ($part->texts as $text) {
230:                 $svg .= $this->renderText($text, $part);
231:             }
232:         }
233:         if ($part->textsOnPath) {
234:             foreach ($part->textsOnPath as $textOnPath) {
235:                 $svg .= $this->renderTextOnPath($textOnPath, $part);
236:             }
237:         }
238:         if (@$part->notes) {
239:             foreach ($part->notes as $note) {
240:                 $svg .= $this->renderNote($note, $part);
241:             }
242:         }
243:         if (@$part->dimensions) {
244:             foreach ($part->dimensions as $dimension) {
245:                 $svg .= $this->renderDimension($dimension, $part);
246:             }
247:         }
248: 
249:         return $svg;
250:     }

281:     /**
282:      * Returns SVG code for text
283:      *
284:      * There's a special attribute 'line-height' that can be set in the
285:      * text attributes. If it's present, it will be used to set the offset
286:      * between lines of text, and somewhat imitating the line-heigt attribute
287:      * of HTML text (SVG has no such thing).
288:      *
289:      * To get multi-line text, just include linebreaks (\n) in your input text.
290:      *
291:      * @param \Freesewing\Text|\Freesewing\TextOnPath $text The Text or TextOnPath to render
292:      * @param \Freesewing\Part $part The part this text is part of
293:      *
294:      * @return string The SVG code for the rendered text
295:      */
296:     private function renderText($text, $part)
297:     {
298:         $anchor = $text->getAnchor();
299:         $svg = $this->nl();
300:         $svg .= '<text x="'.$anchor->getX().'" y="'.$anchor->getY().'" ';
301:         if (!isset($text->attributes['id'])) {
302:             $svg .= 'id="'.$this->getUid().'" ';
303:         }
304:         if (isset($text->attributes['line-height'])) {
305:             $lineHeight = $text->attributes['line-height'];
306:         } else {
307:             $lineHeight = 12;
308:         }
309:         $svg .= Utils::flattenAttributes($text->getAttributes(), ['line-height']);
310:         $svg .= '>';
311: 
312:         $lines = explode("\n", $text->getText());
313:         $attr = '';
314:         $this->indent();
315:         foreach ($lines as $line) {
316:             $svg .= $this->nl()."<tspan $attr>$line</tspan>";
317:             $attr = 'x="'.$anchor->getX().'" dy="'.$lineHeight.'"';
318:         }
319:         $this->outdent();
320:         $svg .= '</text>';
321: 
322:         return $svg;
323:     }

372:     /**
373:      * Returns SVG code for a textOnPath
374:      *
375:      * There's a special attribute 'line-height' that can be set in the
376:      * text attributes. If it's present, it will be used to set the offset
377:      * between lines of text, and somewhat imitating the line-heigt attribute
378:      * of HTML text (SVG has no such thing).
379:      *
380:      * To get multi-line text, just include linebreaks (\n) in your input text.
381:      *
382:      * @param \Freesewing\TextOnPath $textOnPath The textOnPath to render
383:      * @param \Freesewing\Part $part The part this textOnPath is part of
384:      *
385:      * @return string The SVG code for the rendered textOnPath
386:      */
387:     private function renderTextOnPath($textOnPath, $part)
388:     {
389:         $path = $textOnPath->getPath();
390:         $id = $path->getAttribute('id');
391:         // Make sure path has an ID
392:         if(!$id || strlen($id) < 1) {
393:             $id = $this->getUid();
394:             $path->setAttribute('id', $id);
395:         }
396:         $svg = $this->renderPath($path, $part);
397:         $svg .= $this->nl();
398:         $svg .= '<text ';
399:         if (!isset($textOnPath->attributes['id'])) {
400:             $svg .= 'id="'.$this->getUid().'" ';
401:         }
402:         if (isset($textOnPath->attributes['line-height'])) {
403:             $lineHeight = $textOnPath->attributes['line-height'];
404:         } else {
405:             $lineHeight = 12;
406:         }
407:         $svg .= Utils::flattenAttributes($textOnPath->getAttributes(), ['line-height']);
408:         $svg .= '>';
409: 
410:         $svg .= "<textPath xlink:href=\"#$id\" startOffset=\"50%\">".
411:             '<tspan '.Utils::flattenAttributes($textOnPath->getAttributes()).'>'.$textOnPath->getText().'</tspan>'.
412:             '</textPath>';
413:         $svg .= '</text>';
414: 
415:         return $svg;
416:     }
